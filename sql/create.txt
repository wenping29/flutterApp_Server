-- 创建数据库（字符集utf8mb4支持emoji）
CREATE DATABASE IF NOT EXISTS chat_db 
DEFAULT CHARACTER SET utf8mb4 
DEFAULT COLLATE utf8mb4_unicode_ci;

USE chat_db;

-- 用户表
CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE COMMENT '用户名',
    email VARCHAR(100) NOT NULL UNIQUE COMMENT '邮箱',
    password VARCHAR(32) NOT NULL COMMENT '密码（MD5加密）',
    avatar VARCHAR(200) COMMENT '头像URL',
    create_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    last_login_time DATETIME COMMENT '最后登录时间',
    INDEX idx_email (email),
    INDEX idx_username (username)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- 聊天会话表
CREATE TABLE IF NOT EXISTS chat_sessions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    session_id VARCHAR(50) NOT NULL UNIQUE COMMENT '会话唯一标识',
    user_id1 INT NOT NULL COMMENT '参与者1ID',
    user_id2 INT NOT NULL COMMENT '参与者2ID（群聊为0）',
    session_name VARCHAR(50) NOT NULL COMMENT '会话名称',
    last_message VARCHAR(200) COMMENT '最后一条消息',
    last_message_time DATETIME COMMENT '最后消息时间',
    unread_count INT NOT NULL DEFAULT 0 COMMENT '未读消息数',
    is_group BOOLEAN NOT NULL DEFAULT FALSE COMMENT '是否群聊',
    INDEX idx_session_id (session_id),
    INDEX idx_user1 (user_id1),
    INDEX idx_user2 (user_id2)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='聊天会话表';

-- 聊天消息表
CREATE TABLE IF NOT EXISTS chat_messages (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    session_id VARCHAR(50) NOT NULL COMMENT '关联会话ID',
    sender_id INT NOT NULL COMMENT '发送者ID',
    content TEXT NOT NULL COMMENT '消息内容',
    send_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '发送时间',
    is_read BOOLEAN NOT NULL DEFAULT FALSE COMMENT '是否已读',
    INDEX idx_session_id (session_id),
    INDEX idx_sender_id (sender_id),
    INDEX idx_send_time (send_time)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='聊天消息表';

-- 创建收藏表（MySQL）
CREATE TABLE IF NOT EXISTS `Collections` (
  `Id` INT AUTO_INCREMENT PRIMARY KEY COMMENT '收藏ID',
  `UserId` INT NOT NULL COMMENT '用户ID（关联用户表）',
  `Title` VARCHAR(200) NOT NULL COMMENT '收藏标题',
  `Content` VARCHAR(500) DEFAULT '' COMMENT '收藏内容/链接',
  `Type` VARCHAR(50) DEFAULT 'default' COMMENT '收藏类型（文章/工具/链接等）',
  `CreateTime` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '收藏时间',
  -- 索引优化
  INDEX `IX_Collections_UserId` (`UserId`) COMMENT '按用户ID索引'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户收藏表';

-- 创建好友关系表（MySQL）
CREATE TABLE IF NOT EXISTS `Friends` (
  `Id` INT AUTO_INCREMENT COMMENT '主键ID，自增',
  `UserId` INT NOT NULL COMMENT '当前用户ID（谁的好友列表）',
  `FriendUserId` INT NOT NULL COMMENT '好友的用户ID',
  `RemarkName` VARCHAR(50) DEFAULT '' COMMENT '好友备注名',
  `Status` TINYINT NOT NULL DEFAULT 1 COMMENT '好友状态：0-待验证，1-已通过，2-已拉黑',
  `CreateTime` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '添加好友时间',
  -- 主键约束
  PRIMARY KEY (`Id`),
  -- 复合唯一索引：避免同一用户重复添加同一好友
  UNIQUE KEY `UK_User_Friend` (`UserId`, `FriendUserId`),
  -- 普通索引：优化查询效率
  KEY `IX_Friend_Status` (`Status`),
  KEY `IX_Friend_CreateTime` (`CreateTime`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='好友关系表';

-- 创建朋友圈表（MySQL）
CREATE TABLE IF NOT EXISTS `Moments` (
  `Id` INT AUTO_INCREMENT COMMENT '朋友圈ID，自增主键',
  `UserId` INT NOT NULL COMMENT '发布者用户ID（关联Users表）',
  `Content` VARCHAR(500) NOT NULL COMMENT '朋友圈文字内容',
  `Images` VARCHAR(1000) DEFAULT '[]' COMMENT '图片列表（JSON格式，如["url1","url2"]）',
  `CreateTime` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '发布时间',
  `LikeCount` INT NOT NULL DEFAULT 0 COMMENT '点赞数',
  `CommentCount` INT NOT NULL DEFAULT 0 COMMENT '评论数',
  `Visibility` TINYINT NOT NULL DEFAULT 1 COMMENT '可见范围：0-仅自己，1-好友可见，2-公开',
  -- 主键约束
  PRIMARY KEY (`Id`),
  -- 索引优化（适配高频查询场景）
  KEY `IX_Moments_UserId` (`UserId`) COMMENT '按发布者ID索引，优化个人朋友圈查询',
  KEY `IX_Moments_CreateTime` (`CreateTime`) COMMENT '按发布时间索引，优化按时间排序',
  KEY `IX_Moments_Visibility` (`Visibility`) COMMENT '按可见范围索引，优化权限筛选'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='朋友圈动态表';
